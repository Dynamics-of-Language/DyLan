event_restrictor(past)
	put(ty(es>(es>cn)))
	put(fo(X^Y^[p:(Y subset X & X<now)]))

event_restrictor(pres)
	put(ty(es>(es>cn)))
	put(fo(X^Y^[p:(Y overlap X & X=now)]))

event_restrictor(future)
	put(ty(es>(es>cn)))
	put(fo(X^Y^[p:(now<X)]))

event_restrictor(inf)

	go(/\0\/1) *//

event_restrictor(bare)
	put(?+restrictor)
	
inf_event_restrictor(pres)
