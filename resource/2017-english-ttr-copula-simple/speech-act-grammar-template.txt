info-color
IF	  	¬ty(Y>Z)
		ty(X)
		¬</\0\/1>Ex.x
		¬</\L>Ex.x
		¬?+eval
		Fo(W1)
		W1 << [...]
		PrevSpeaker(V)
THEN	IF		<Z>+Q
		THEN	abort
		ELSE	IF		<Z>+neg
				THEN	abort
				ELSE	put(sa:info(V,color:%colorvalue))
ELSE	abort
